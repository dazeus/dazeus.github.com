00062 =head2 connect($socket)
00063 
00064 Creates a DaZeus object connected to the given socket. Returns the object if
00065 the initial connection succeeded; otherwise, calls die(). If, after this
00066 initial connection, the connection fails, for example because the bot is
00067 restarted, the module will re-connect.
00068 
00069 The given socket name must either start with "tcp:" or "unix:"; in the case of
00070 a UNIX socket it must contain a path to the UNIX socket, in the case of TCP it
00071 may end in a port number. IPv6 addresses must have a port number attached, so
00072 it can be distinguished from the rest of the address; in "tcp:2001:db8::1:1234"
00073 1234 is the port number.
00074 
00075 If you give an object through $socket, the module will attempt to use it
00076 directly as the target socket.
00077

00400 sub setProperty {
00401         my ($self, $name, $value, @scope) = @_;
00402         if(!ref($value)) {
00403                 # Storable has provisions for storing additional scalar state, such
00404                 # as the UTF-8 flag (see perlunicode). Therefore, even simple strings

00500 =head2 handleEvent($timeout)
00501 
00502 Returns all events that can be returned as soon as possible, but no longer
00503 than the given $timeout. If $timeout is zero, do not block. If timeout is
00504 undefined, wait forever until the first event arrives.
00505 
00506 "As soon as possible" means that if there are cached events already read from
00507 the socket earlier, the socket will not be touched. Otherwise, if events are
00508 available for reading, they will be immediately read. Only if no events were
00509 cached, none were available for reading, and $timeout is not zero will this
00510 function block to retrieve events. (See also handleEvents().)
00511 
00512 This method only returns a single event, in order of receiving. For every
00513 returned event, the event handler (if given to subscribe()), is called.
00514 
00515 =cut
00516 
00517 sub handleEvent {
00518         my ($self, $timeout) = @_;
00519         if(@{$self->{events}} == 0) {
00520                 my $packet = $self->_readPacket($timeout);
00521                 if(!$packet) {
00522                         return;
00523                 }
00524                 if(!$packet->{event}) {
00525                         die "Error: Out of bound non-event packet received in handleEvent";
00526                 }
00527                 push @{$self->{events}}, $packet;
00528         }
00529         my $event = shift @{$self->{events}};
00530         my $handler = $self->{handlers}{$event->{event}};
00531         if($handler) {
00532                 $handler->($self, $event);
00533         }
00534         return $event;
00535 }

00537 sub _readPacket {
00538         use bytes;
00539         my ($self, $timeout) = @_;
00540         my $once = $timeout == 0 if(defined $timeout);
00541         my $stop_time = time() + $timeout if(defined $timeout);
00542         my $event_part;
00543         my $last_event_part;
00544         my $size = 3;
00545         # A size of 20 bytes is too large, something must be happening
00546         while($size < 20 && ($once || !defined($stop_time) || time() < $stop_time)) {
00547                 undef $event_part;
00548                 if(!defined($self->{sock}->recv($event_part, $size, MSG_PEEK))) {
00549                         if($! == EAGAIN) {
00550                                 # Nothing to read, that's ok
00551                                 if(defined $timeout && $timeout == 0) {
00552                                         return;
00553                                 }
00554                                 # TODO: how to set a timeout parameter on sockets after creation?
00555                                 next;
00556                         }
00557                         # Error in peek
00558                         delete $self->{sock};
00559                         die $!;
00560                 }
00561                 # On OS X, if the other end closes the socket, recv() still returns success
00562                 # values and sets no error; however, connected() will be undef there
00563                 if(!defined($self->{sock}->connected())) {
00564                         # Error in peek
00565                         delete $self->{sock};
00566                         die $!;
00567                 }
00568                 # TODO: change this to "if there are no more bytes available", more accurate
00569                 if($last_event_part && $last_event_part eq $event_part) {
00570                         $once = 0;
00571                 }
00572                 $last_event_part = $event_part;
00573                 if($event_part =~ /^[\r\n]+$/) {
00574                         # This seems to be necessary on my Linux machine; it refuses
00575                         # to read further than these newline bytes...
00576                         $self->{sock}->recv($event_part, length $event_part);
00577                 }
00578                 # Has enough data been read?
00579                 if(length($event_part) == $size && index($event_part, '{') > 0) {
00580                         # Start of JSON data found
00581                         my $pre_json_size = index($event_part, '{');
00582                         my $json_size = substr($event_part, 0, $pre_json_size);
00583                         # Ignore all non-numerals:
00584                         $json_size =~ s/[^0-9]//g;
00585 
00586                         # See if we can pull all the data from the socket now
00587                         if(!defined($self->{sock}->recv($event_part, $pre_json_size + $json_size, MSG_PEEK))) {
00588                                 # Error in peek
00589                                 delete $self->{sock};
00590                                 die $!;
00591                         }
00592                         if(length($event_part) != ($pre_json_size + $json_size)) {
00593                                 # Try again later
00594                                 next;
00595                         }
00596                         # That worked, so now we can actually take it off the socket
00597                         if(!defined($self->{sock}->recv($event_part, $pre_json_size + $json_size))) {
00598                                 # Error in read
00599                                 delete $self->{sock};
00600                                 die $!;
00601                         }
00602                         my $json = decode_json(substr($event_part, $pre_json_size));
00603                         return $json;
00604                 } elsif(length($event_part) == $size) {
00605                         # We read something, but it's not in this response yet
00606                         $size += 4;
00607                 }
00608         }
00609         return;
00610 }
