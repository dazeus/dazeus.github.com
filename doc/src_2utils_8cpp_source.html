00035 uint8_t read_utf8_byte(const std::string &json, size_t &pos, bool is_first) {
00036         if(json[pos] != '\\' || json[pos+1] != 'u') {
00037                 throw utf8_encoding_error("Invalid character found where new UTF-8 byte expected");
00038         }
00039         std::stringstream ss;
00040         char unicode[4] = {json[pos+2], json[pos+3], json[pos+4], json[pos+5]};
00041         pos += 6;
00042         ss << std::hex << unicode;
00043         uint32_t res;
00044         ss >> res;
00045 
00047         // If it's 0xxxxxxx, it's a single ASCII byte, only allowed if it's the "first" one
00048         if(is_first && res < 128) {
00049                 return (uint8_t)res;
00050         }
00051         // We're fixing issues where \u00XX appears incorrectly, assuming this is needed
00052         if(res > 255) {
00053                 throw utf8_encoding_error(res, "JSON writer wrote characters with codepoint above byte max");
00054         }
00055         // If it's a first byte, may not be 10xxxxxx
00056         if(is_first && res < 192) {
00057                 throw utf8_encoding_error(res, "First bits of Unicode first byte must be 11");
00058         }
00059         // If it's a continuation byte, must be 10xxxxxx
00060         if(!is_first && (res < 128 || res >= 192)) {
00061                 throw utf8_encoding_error(res, "First bits of Unicode continuation byte must be 10");
00062         }
00063 
00064         return (uint8_t)res;
00065 }

00067 uint32_t read_utf8_codepoint(const std::string &json, size_t &pos) {
00068         std::vector<uint8_t> replaced_bytes;
00069         uint8_t first = read_utf8_byte(json, pos, true);
00070         if(first < 128) {
00071                 // Single byte, leave as-is
00072                 return first;
00073         }
00074         replaced_bytes.push_back(first);
00075 
00076         // Source: https://en.wikipedia.org/wiki/Utf-8#Description
00077         // The number of bytes for the character is the number of set bits to
00078         // the left of this byte; i.e. 0b1111_0XXX means this character
00079         // consists of 4 bytes -- except for a character consisting of 1 byte,
00080         // but that's already handled above.
00081         uint8_t num_bytes;
00082         for(num_bytes = 0;; ++num_bytes) {
00083                 // explictly store this in a variable, as the compiler might
00084                 // optimize (first << 2 >> 7) to be (first >> 5).
00085                 uint8_t s = first << num_bytes;
00086                 if((s >> 7) == 0) break;
00087         }
00088 
00089         // Since RFC 3629, UTF-8 won't have > 4 characters anymore, and we
00090         // won't handle 1-byte characters here.
00091         if(num_bytes == 1 || num_bytes > 4) {
00092                 std::cerr << "In read_utf8_codepoint(), JSON string has invalid UTF-8 "
00093                              "character (num_bytes = " << num_bytes << "), returning as-is."
00094                           << std::endl;
00095                 return first;
00096         }
00097 
00098         // Assemble the resulting codepoint using the remaining bits
00099         uint32_t res;
00100         {
00101                 uint32_t s = first;
00102                 s = s << (num_bytes + 25);
00103                 res = s >> (num_bytes + 25);
00104         }
00105         while(num_bytes-- > 1) {
00106                 try {
00107                         uint8_t next_byte = read_utf8_byte(json, pos, false);
00108                         replaced_bytes.push_back(next_byte);
00109                         uint8_t s = next_byte << 2;
00110                         // Six new bytes!
00111                         res = (res << 6) + (s >> 2);
00112                 } catch(utf8_encoding_error &e) {
00113                         if(e.original_bytes.size() > 0) {
00114                                 assert(e.original_bytes.size() == 1);
00115                                 replaced_bytes.insert(replaced_bytes.end(), e.original_bytes.begin(), e.original_bytes.end());
00116                         }
00117                         throw utf8_encoding_error(replaced_bytes, e.what());
00118                 }
00119         }
00120 
00121         return res;
00122 }

00132 std::string fix_unicode_in_json(const std::string &json) {
00133         // This function is intended to fix UTF8 encoding in the given JSON
00134         // string. In a std::string, all characters are assumed to be 1 byte,
00135         // but in UTF8 characters can span any number of bytes between 1 and 6.
00136         // When libjson converts a std::string to JSON, it takes every byte
00137         // and transforms it into its own \uXXXX character, i.e. the character
00138         // \u0192 becomes \u00c6\u0092. This function fixes that.
00139         std::string fixed;
00140         bool escaped = false;
00141         for(size_t i = 0; i < json.length(); ++i) {
00142                 unsigned char t = json[i];
00143                 if(escaped) {
00144                         escaped = false;
00145                         if(t == 'u') { // beginning of Unicode character
00146                                 try {
00147                                         i--; // read_utf8_codepoint starts at \, we're already at u
00148                                         uint32_t cp_value = read_utf8_codepoint(json, i);
00149                                         fixed += 'u'; // the slash was already copied
00150                                         fixed += cp_to_hex(cp_value);
00151                                 } catch(utf8_encoding_error &e) {
00152                                         // Not an Unicode encoding, write the original bytes
00153                                         assert(e.original_bytes.length() >= 1);
00154                                         bool first = true;
00155                                         for(size_t i = 0; i < e.original_bytes.size(); ++i) {
00156                                                 uint8_t cp = e.original_bytes[i];
00157                                                 if(!first) fixed += '\\';
00158                                                 fixed += 'u';
00159                                                 fixed += cp_to_hex(cp);
00160                                                 first = false;
00161                                         }
00162                                 }
00163                                 i--; // read_utf8_codepoint ends at next char
00164                         } else {
00165                                 fixed += t;
00166                         }
00167                 } else {
00168                         fixed += t;
00169                         if(t == '\\') {
00170                                 escaped = true;
00171                         }
00172                 }
00173         }
00174         return fixed;
00175 }
